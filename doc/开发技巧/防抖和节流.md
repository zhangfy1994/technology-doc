## 防抖

```js
function debounce(fn, time) {
  let timer;
  const that = this;
  return function (...params) {
    if (timer) {
      clearTimeout(timer);
    }

    timer = setTimeout(() => {
      fn.apply(that, params);
    }, time);
  };
}
```

## 节流

```js
function throttle(fn, time) {
  let canExecute = true;

  return function (...params) {
    const that = this;
    if (canExecute) {
      canExecute = false;
      setTimeout(() => {
        fn.apply(that, params);
        canExecute = true;
      }, time);
    }
  };
}
```
